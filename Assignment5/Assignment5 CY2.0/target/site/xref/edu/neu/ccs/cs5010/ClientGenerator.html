11  public class ClientGenerator {
12      public static final int clientIDBound = 1000000;
13      public static final int depositBound = 2000;
14      public static final int withdrawalBound = 3000;
15  
16      public HashMap<Integer, Client> clientData;
17      public int clientNumber;
18  
19      /**
20       * Client data constructor.
21       */
22      public ClientGenerator(int clientNum){
23          this.clientNumber = clientNum;
24      }
25  
26      /**
27       * Generate the client data.
28       */
29      public HashMap<Integer, Client> generateClientData(){
30          RandomNumber randomNum = new RandomNumber();
31          HashSet<Integer> clientIDSet = new HashSet<>();
32          RSAKeyGenerator keyGenerator = new RSAKeyGenerator();
33          Client client;
34          BigInteger[][] keyPair;
35          BigInteger[] publicKey;
36          BigInteger[] privateKey;
37  
38          int clientID;
39          clientData = new HashMap();
40          for (int i = 0; i < this.clientNumber; ) {
41              clientID = randomNum.getRandom(clientIDBound) + 1;
42              if (!clientIDSet.contains(clientID)) {
43                  keyPair = keyGenerator.generateKey();
44                  publicKey = keyPair[RSAKeyGenerator.publicKeyIndex];
45                  privateKey = keyPair[RSAKeyGenerator.privateKeyIndex];
46                  client = new Client(clientID, publicKey, privateKey,
47                          randomNum.getRandom(depositBound + 1), randomNum.getRandom(withdrawalBound + 1));
48                  clientData.put(clientID, client);
49                  i++;
50              }
51          }
52          return this.clientData;
53      }
54  }
