12  public class SecureBankVerificationSimulator {
13  
14      public static final int clientsNumberIndex = 0;
15      public static final int verificationNumbertIndex = 1;
16      public static final int fractionIndex = 2;
17  
18      private static HashMap<Integer, Client> clientData;
19      private static HashMap<Integer, Client> clientBankData;
20      private static ArrayList<BigInteger[]> rawPairs;
21      private static ArrayList<Integer> clientIDArray;
22  
23      /**
24       * Generate and store all clients, both clients itself and the clients in bank.
25       * Also, store all the ID information in an array.
26       *
27       */
28      public static void generateClients(int clientsNumber) {
29          ClientGenerator clientGenerator = new ClientGenerator(clientsNumber);
30          clientData = clientGenerator.generateClientData();
31          Client client;
32          for (Map.Entry<Integer, Client> entry: clientData.entrySet()) {
33              client = entry.getValue();
34              client.deletePrivateKey();
35              clientBankData.put(entry.getKey(), client);
36              clientIDArray.add(client.getClientID());
37          }
38  
39      }
40  
41      /**
42       * Generate message & signature pairs
43       */
44      public static void generatePairs(int pairNumber, float fraction) {
45          BigInteger[] pair;
46          Client client = null;
47          PairGenerator pairGenerator = new PairGenerator();
48          for (int i = 0; i < pairNumber; i++) {
49              pair = pairGenerator.generatePairs(fraction, client);
50              rawPairs.add(pair);
51          }
52      }
53  
54      /**
55       * Read command line and set the neighbor.
56       * add all the candy from historical data
57       */
58      public static void simulate(String[] commandLine) {
59          int clientsNumber = Integer.parseInt(commandLine[clientsNumberIndex]);
60          int pairNumber = Integer.parseInt(commandLine[verificationNumbertIndex]);
61          float fraction = Integer.parseInt(commandLine[fractionIndex]);
62          generateClients(clientsNumber);
63          generatePairs(pairNumber, fraction);
64  
65      }
66  
67      /**
68       * Main method, get command line from the user.
69       */
70      public static void main(String[] args) {
71          String[] testLine = {"200", "200", "0.2", "output.txt"};
72          simulate(testLine);
73      }
