20       * @return the random generated patient
21       * given a start time and duration, generate a time that is start time + random(0, duration)
22       */
23      public Patient newPatient(LocalDateTime localTime, int duration) throws Exception {
24          Random random = new Random();
25          int Urgency = random.nextInt(Patient.MAX_Urgency) + 1;
26          int RandomTreatment = random.nextInt(Patient.MAX_Treatment) + 1;
27          int minuteDuration = random.nextInt((duration )) + 1;
28  
29          LocalDateTime RandomTime = localTime.plusMinutes(minuteDuration);
30          Patient RandomPatient = new Patient(Urgency, RandomTime, RandomTreatment);
31          return RandomPatient;
32      }
