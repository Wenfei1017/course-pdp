3   import java.io.IOException;
4   import java.util.ArrayList;

7    * Created by wenfei on 10/23/17.

9   public class FileWriter {
10  
11      /**
12       * Get the current candy list and file name, create the file.
13       *
14       * @param currentCandy the fileName
15       * @param fileName the fileName
16       *
17       * @throws IllegalArgumentException if file is not created successfully
18       */
19      public void writeFile(ArrayList<Candy> currentCandy, String fileName) throws IllegalArgumentException {
20          if (fileName == null || currentCandy == null) {
21              throw new IllegalArgumentException("filename and candylist should not be null");
22          }
23          java.io.FileWriter Writer = null;
24  
25          try {
26              Writer = new java.io.FileWriter(fileName.replace("Candy", "Travesal"));
27              Writer.write("Candy Type, Candy size, Household type\n");
28              for (Candy candy: currentCandy) {
29                  Writer.write(candy.getCandySize() + " Size, ");
30                  Writer.write(candy.getCandyName() + ", ");
31                  Writer.write(candy.getCandyHouse() + "\n");
32              }
33          } catch(IOException e) {
34              e.getStackTrace();
35          } finally {
36              if (Writer != null) {
37                  try {
38                      Writer.close();
39                  } catch (IOException e) {
40                      e.getStackTrace();
41                  }
42              }
43          }
44      }
45  }
